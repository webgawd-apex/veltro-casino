pub fn process_instruction(
    _program_id: &Pubkey,

    let mut accounts_iter = accounts.iter();

    let user_account = next_account_info(&mut accounts_iter)?;
    let program_id = next_account_info(&mut accounts_iter)?;

    // Check if the token account belongs to the user
    if token_account.owner != user_account.key {
        return Err(ProgramError::InvalidAccountData);

    // Serialize back to account
    TokenAccount::pack(token_account_state, &mut token_data)?;

    msg!("Token approval completed");
    Ok(())
}

fn approve_token(accounts: &[AccountInfo]) -> ProgramResult {
